    62→function RAGSparkline({ cycleCount }: { cycleCount: number }) {
    63→  const canvasRef = useRef<HTMLCanvasElement>(null);
    64→
    65→  useEffect(() => {
    66→    const canvas = canvasRef.current;
    67→    if (!canvas) return;
    68→    const ctx = canvas.getContext('2d');
    69→    if (!ctx) return;
    70→
    71→    const rect = canvas.getBoundingClientRect();
    72→    const dpr = window.devicePixelRatio || 1;
    73→    canvas.width = rect.width * dpr;
    74→    canvas.height = rect.height * dpr;
    75→    ctx.scale(dpr, dpr);
    76→
    77→    const w = rect.width;
    78→    const h = rect.height;
    79→
    80→    // Generate 20 synthetic performance values based on cycle count
    81→    const data: number[] = [];
    82→    for (let i = 0; i < 20; i++) {
    83→      const base = 75 + cycleCount * 2;
    84→      const noise = Math.sin(i * 0.8 + cycleCount * 1.5) * 8 + Math.cos(i * 1.3) * 5;
    85→      data.push(Math.max(50, Math.min(100, base + noise)));
    86→    }
    87→
    88→    const padX = 4;
    89→    const padY = 4;
    90→    const chartW = w - padX * 2;
    91→    const chartH = h - padY * 2;
    92→    const minVal = 50;
    93→    const maxVal = 100;
    94→    const range = maxVal - minVal;
    95→
    96→    // Gradient fill
    97→    const grad = ctx.createLinearGradient(0, padY, 0, h);
    98→    grad.addColorStop(0, 'rgba(168, 85, 247, 0.3)');
    99→    grad.addColorStop(1, 'rgba(168, 85, 247, 0.0)');
   100→
   101→    ctx.beginPath();
   102→    data.forEach((val, i) => {
   103→      const x = padX + (i / (data.length - 1)) * chartW;
   104→      const y = padY + chartH - ((val - minVal) / range) * chartH;
   105→      if (i === 0) ctx.moveTo(x, y);
   106→      else ctx.lineTo(x, y);
   107→    });
   108→
   109→    // Close path for fill
   110→    ctx.lineTo(padX + chartW, padY + chartH);
   111→    ctx.lineTo(padX, padY + chartH);
   112→    ctx.closePath();
   113→    ctx.fillStyle = grad;
   114→    ctx.fill();
   115→
   116→    // Line
   117→    ctx.beginPath();
   118→    data.forEach((val, i) => {
   119→      const x = padX + (i / (data.length - 1)) * chartW;
   120→      const y = padY + chartH - ((val - minVal) / range) * chartH;
   121→      if (i === 0) ctx.moveTo(x, y);
   122→      else ctx.lineTo(x, y);
   123→    });
   124→    ctx.strokeStyle = '#a855f7';
   125→    ctx.lineWidth = 1.5;
   126→    ctx.stroke();
   127→
   128→    // Last point dot
   129→    const lastX = padX + chartW;
   130→    const lastY = padY + chartH - ((data[data.length - 1] - minVal) / range) * chartH;
   131→    ctx.beginPath();
   132→    ctx.arc(lastX, lastY, 2.5, 0, Math.PI * 2);
   133→    ctx.fillStyle = '#a855f7';
   134→    ctx.fill();
   135→    ctx.beginPath();
   136→    ctx.arc(lastX, lastY, 5, 0, Math.PI * 2);
   137→    ctx.fillStyle = 'rgba(168, 85, 247, 0.25)';
   138→    ctx.fill();
   139→  }, [cycleCount]);
   140→
   141→  return (
   142→    <div className="flex items-center gap-2">
   143→      <span className="text-[9px] font-mono text-[#8888aa] uppercase tracking-wider">RAG Perf</span>
   144→      <canvas
   145→        ref={canvasRef}
   146→        className="block"
   147→        style={{ width: '120px', height: '36px' }}
   148→      />
   149→    </div>
   150→  );
   151→}
