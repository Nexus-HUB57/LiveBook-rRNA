   104тЖТ   104тЖТfunction drawRoundedRect(
   105тЖТ   105тЖТ  ctx: CanvasRenderingContext2D,
   106тЖТ   106тЖТ  x: number, y: number, w: number, h: number, r: number
   107тЖТ   107тЖТ) {
   108тЖТ   108тЖТ  ctx.moveTo(x + r, y);
   109тЖТ   109тЖТ  ctx.lineTo(x + w - r, y);
   110тЖТ   110тЖТ  ctx.arcTo(x + w, y, x + w, y + r, r);
   111тЖТ   111тЖТ  ctx.lineTo(x + w, y + h - r);
   112тЖТ   112тЖТ  ctx.arcTo(x + w, y + h, x + w - r, y + h, r);
   113тЖТ   113тЖТ  ctx.lineTo(x + r, y + h);
   114тЖТ   114тЖТ  ctx.arcTo(x, y + h, x, y + h - r, r);
   115тЖТ   115тЖТ  ctx.lineTo(x, y + r);
   116тЖТ   116тЖТ  ctx.arcTo(x, y, x + r, y, r);
   117тЖТ   117тЖТ  ctx.closePath();
   118тЖТ   118тЖТ}

   276тЖТ   276тЖТ        n.vy += (targetY - n.y) * 0.00005;

   602тЖТ   602тЖТ          const glowGrad = ctx.createRadialGradient(n.x, n.y, drawSize, n.x, n.y, drawSize + 15);
   603тЖТ   603тЖТ          glowGrad.addColorStop(0, n.color + '30');
   604тЖТ   604тЖТ          glowGrad.addColorStop(1, n.color + '00');
   605тЖТ   605тЖТ          ctx.fillStyle = glowGrad;
   606тЖТ   606тЖТ          ctx.fill();
   607тЖТ   607тЖТ        }
   608тЖТ   608тЖТ
   609тЖТ   609тЖТ        // Node circle
   610тЖТ   610тЖТ        ctx.beginPath();
   611тЖТ   611тЖТ        ctx.arc(n.x, n.y, drawSize, 0, Math.PI * 2);
   612тЖТ   612тЖТ        const nodeGrad = ctx.createRadialGradient(n.x - drawSize * 0.3, n.y - drawSize * 0.3, 0, n.x, n.y, drawSize);
   613тЖТ   613тЖТ        nodeGrad.addColorStop(0, n.color + (n.active ? 'ee' : isHovered ? 'cc' : '88'));
   614тЖТ   614тЖТ        nodeGrad.addColorStop(1, n.color + (n.active ? '44' : '22'));
   615тЖТ   615тЖТ        ctx.fillStyle = nodeGrad;
   616тЖТ   616тЖТ        ctx.fill();
   617тЖТ   617тЖТ        ctx.strokeStyle = n.color + (n.active ? 'ff' : isHovered ? 'aa' : '55');
   618тЖТ   618тЖТ        ctx.lineWidth = n.active ? 2 : 1;
   619тЖТ   619тЖТ        ctx.stroke();
   620тЖТ   620тЖТ
   621тЖТ   621тЖТ        // Label
   622тЖТ   622тЖТ        ctx.fillStyle = n.active || isHovered ? '#ffffff' : '#c0b8d0';

   624тЖТ   624тЖТ        ctx.textAlign = 'center';
   625тЖТ   625тЖТ        ctx.textBaseline = 'top';
   626тЖТ   626тЖТ        ctx.fillText(n.label, n.x, n.y + drawSize + 5);
   627тЖТ   627тЖТ
   628тЖТ   628тЖТ        // ---- Sparkline (activity history) below label ----
   629тЖТ   629тЖТ        const sparklineY = n.y + drawSize + 5 + Math.max(8, n.size * 0.55) + 4;
   630тЖТ   630тЖТ        const sparklineW = n.size * 2.2;
   631тЖТ   631тЖТ        const sparklineH = 6;
   632тЖТ   632тЖТ        const sparklineX = n.x - sparklineW / 2;

   638тЖТ   638тЖТ            const hy = sparklineY + sparklineH - history[hi] * sparklineH;
   639тЖТ   639тЖТ            if (hi === 0) ctx.moveTo(hx, hy);
   640тЖТ   640тЖТ            else ctx.lineTo(hx, hy);
   641тЖТ   641тЖТ          }

   646тЖТ   646тЖТ
   647тЖТ   647тЖТ          // Sparkline fill
   648тЖТ   648тЖТ          ctx.lineTo(sparklineX + sparklineW, sparklineY + sparklineH);
   649тЖТ   649тЖТ          ctx.lineTo(sparklineX, sparklineY + sparklineH);
   650тЖТ   650тЖТ          ctx.closePath();
   651тЖТ   651тЖТ          ctx.fillStyle = n.color + Math.floor(sparkAlpha * 0.3 * 255).toString(16).padStart(2, '0');
   652тЖТ   652тЖТ          ctx.fill();
   653тЖТ   653тЖТ        }
   654тЖТ   654тЖТ      });
   655тЖТ   655тЖТ
   656тЖТ   656тЖТ      setHoveredNode(hoveredRef);
   657тЖТ   657тЖТ      animRef.current = requestAnimationFrame(draw);
   658тЖТ   658тЖТ    };
   659тЖТ   659тЖТ
   660тЖТ   660тЖТ    draw();
   661тЖТ   661тЖТ
   662тЖТ   662тЖТ    return () => {
   663тЖТ   663тЖТ      cancelAnimationFrame(animRef.current);
   664тЖТ   664тЖТ      window.removeEventListener('resize', resize);
   665тЖТ   665тЖТ      canvas.removeEventListener('mousemove', handleMouseMove);

   669тЖТ   669тЖТ    };
   670тЖТ   670тЖТ  }, [isActive, activeNode, onNodeClick]);
   671тЖТ   671тЖТ

   673тЖТ   673тЖТ    <canvas
   674тЖТ   674тЖТ      ref={canvasRef}

   677тЖТ   677тЖТ    />
   678тЖТ   678тЖТ  );
