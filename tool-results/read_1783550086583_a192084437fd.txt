    89→function LatencyChart({ cores }: { cores: CoreStatus[] }) {
    90→  const canvasRef = useRef<HTMLCanvasElement>(null);
    91→  const historyRef = useRef<Record<CoreId, number[]>>({
    92→    ollama: [],
    93→    llama4: [],
    94→    openai: [],
    95→  });
    96→
    97→  useEffect(() => {
    98→    cores.forEach(c => {
    99→      if (c.status !== 'offline' && c.status !== 'loading' && c.latency > 0) {
   100→        historyRef.current[c.id].push(c.latency);
   101→        if (historyRef.current[c.id].length > 30) {
   102→          historyRef.current[c.id] = historyRef.current[c.id].slice(-30);
   103→        }
   104→      }
   105→    });
   106→
   107→    const canvas = canvasRef.current;
   108→    if (!canvas) return;
   109→    const ctx = canvas.getContext('2d');
   110→    if (!ctx) return;
   111→
   112→    const rect = canvas.getBoundingClientRect();
   113→    const dpr = window.devicePixelRatio || 1;
   114→    canvas.width = rect.width * dpr;
   115→    canvas.height = rect.height * dpr;
   116→    ctx.scale(dpr, dpr);
   117→
   118→    const w = rect.width;
   119→    const h = rect.height;
   120→    const padX = 2;
   121→    const padY = 2;
   122→    const chartW = w - padX * 2;
   123→    const chartH = h - padY * 2;
   124→
   125→    ctx.clearRect(0, 0, w, h);
   126→
   127→    // Find max value across all histories
   128→    let maxVal = 50;
   129→    Object.values(historyRef.current).forEach(hist => {
   130→      hist.forEach(v => { if (v > maxVal) maxVal = v; });
   131→    });
   132→    maxVal = Math.ceil(maxVal / 20) * 20;
   133→
   134→    const coreColors: Record<CoreId, string> = {
   135→      ollama: '#06d6a0',
   136→      llama4: '#fbbf24',
   137→      openai: '#e040a0',
   138→    };
   139→
   140→    (Object.keys(historyRef.current) as CoreId[]).forEach(coreId => {
   141→      const data = historyRef.current[coreId];
   142→      if (data.length < 2) return;
   143→
   144→      ctx.beginPath();
   145→      data.forEach((val, i) => {
   146→        const x = padX + (i / 29) * chartW;
   147→        const y = padY + chartH - (val / maxVal) * chartH;
   148→        if (i === 0) ctx.moveTo(x, y);
   149→        else ctx.lineTo(x, y);
   150→      });
   151→      ctx.strokeStyle = coreColors[coreId];
   152→      ctx.lineWidth = 1.2;
   153→      ctx.globalAlpha = 0.8;
   154→      ctx.stroke();
   155→      ctx.globalAlpha = 1;
   156→    });
   157→
   158→    // Y-axis labels
   159→    ctx.font = '7px monospace';
   160→    ctx.fillStyle = '#555577';
   161→    ctx.textAlign = 'right';
   162→    ctx.textBaseline = 'top';
   163→    ctx.fillText(`${maxVal}`, w - 2, padY);
   164→    ctx.fillText('0', w - 2, padY + chartH - 8);
   165→  }, [cores]);
   166→
   167→  return (
   168→    <div className="col-span-2 sm:col-span-2">
   169→      <canvas
   170→        ref={canvasRef}
   171→        className="block w-full"
   172→        style={{ height: '40px' }}
   173→      />
   174→    </div>
   175→  );
   176→}
