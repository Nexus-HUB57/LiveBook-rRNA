     1→'use client';
     2→
     3→import { useEffect, useRef } from 'react';
     4→
     5→interface CoreDef {
     6→  id: 'ollama' | 'llama4' | 'openai';
     7→  label: string;
     8→  sublabel: string;
     9→  color: string;
    10→  glowColor: string;
    11→  angle: number;
    12→}
    13→
    14→const CORES: CoreDef[] = [
    15→  { id: 'ollama', label: 'Ollama', sublabel: 'Local Runtime', color: '#06d6a0', glowColor: 'rgba(6,214,160,', angle: -Math.PI / 2 },
    16→  { id: 'llama4', label: 'Llama 4', sublabel: 'Maverick', color: '#fbbf24', glowColor: 'rgba(251,191,36,', angle: Math.PI / 6 },
    17→  { id: 'openai', label: 'OpenAI', sublabel: 'Native API', color: '#e040a0', glowColor: 'rgba(224,64,160,', angle: (5 * Math.PI) / 6 },
    18→];
    19→
    20→interface TrinuclearSandboxCanvasProps {
    21→  isActive: boolean;
    22→  corePower: Record<string, number>;
    23→  syncIntensity: number;
    24→}
    25→
    26→interface EnergyParticle {
    27→  x: number;
    28→  y: number;
    29→  targetCore: number;
    30→  sourceCore: number;
    31→  progress: number;
    32→  speed: number;
    33→  color: string;
    34→  size: number;
    35→  trail: { x: number; y: number }[];
    36→}
    37→
    38→interface PulseWave {
    39→  x: number;
    40→  y: number;
    41→  radius: number;
    42→  maxRadius: number;
    43→  color: string;
    44→  alpha: number;
    45→}
    46→
    47→interface AmbientParticle {
    48→  x: number;
    49→  y: number;
    50→  vx: number;
    51→  vy: number;
    52→  size: number;
    53→  alpha: number;
    54→}
    55→
    56→interface CoreThroughput {
    57→  values: number[];
    58→}
    59→
    60→interface NeuralMeshNode {
    61→  x: number;
    62→  y: number;
    63→  connections: number[];
    64→  activation: number;
    65→}
    66→
    67→interface CoherenceArc {
    68→  startAngle: number;
    69→  endAngle: number;
    70→  radius: number;
    71→  coherence: number;
    72→  color: string;
    73→}
    74→
    75→export default function TrinuclearSandboxCanvas({ isActive, corePower, syncIntensity }: TrinuclearSandboxCanvasProps) {
    76→  const canvasRef = useRef<HTMLCanvasElement>(null);
    77→  const particlesRef = useRef<EnergyParticle[]>([]);
    78→  const mouseRef = useRef({ x: -1000, y: -1000 });
    79→  const animRef = useRef<number>(0);
    80→  const pulseWavesRef = useRef<PulseWave[]>([]);
    81→  const ambientParticlesRef = useRef<AmbientParticle[]>([]);
    82→  const coreThroughputRef = useRef<CoreThroughput[]>(
    83→    CORES.map(() => ({ values: [0, 0, 0, 0, 0] }))
    84→  );
    85→  const prevPowerRef = useRef<Record<string, number>>({});
    86→  const coreSatellitesRef = useRef<{ angle: number; speed: number; distance: number }[][]>(
    87→    CORES.map(() => {
    88→      const sats: { angle: number; speed: number; distance: number }[] = [];
    89→      for (let i = 0; i < 3; i++) {
    90→        sats.push({
    91→          angle: (Math.PI * 2 / 3) * i,
    92→          speed: 0.8 + Math.random() * 0.6,
    93→          distance: 42 + Math.random() * 8,
    94→        });
    95→      }
    96→      return sats;
    97→    })
    98→  );
    99→  const neuralMeshRef = useRef<NeuralMeshNode[]>([]);
   100→  const coherenceArcsRef = useRef<CoherenceArc[]>([]);
   101→
   102→  useEffect(() => {
   103→    const canvas = canvasRef.current;
   104→    if (!canvas) return;
   105→    const ctx = canvas.getContext('2d');
   106→    if (!ctx) return;
   107→
   108→    const resize = () => {
   109→      const rect = canvas.getBoundingClientRect();
   110→      const dpr = window.devicePixelRatio || 1;
   111→      canvas.width = rect.width * dpr;
   112→      canvas.height = rect.height * dpr;
   113→      ctx.scale(dpr, dpr);
   114→    };
   115→    resize();
   116→    window.addEventListener('resize', resize);
   117→
   118→    const handleMouseMove = (e: MouseEvent) => {
   119→      const rect = canvas.getBoundingClientRect();
   120→      mouseRef.current = { x: e.clientX - rect.left, y: e.clientY - rect.top };
   121→    };
   122→    canvas.addEventListener('mousemove', handleMouseMove);
   123→
   124→    let time = 0;
   125→    let lastSpawn = 0;
   126→    let lastThroughputUpdate = 0;
   127→
   128→    // Initialize ambient particles
   129→    if (ambientParticlesRef.current.length === 0) {
   130→      for (let i = 0; i < 40; i++) {
   131→        ambientParticlesRef.current.push({
   132→          x: Math.random() * 2000,
   133→          y: Math.random() * 2000,
   134→          vx: (Math.random() - 0.5) * 0.15,
   135→          vy: (Math.random() - 0.5) * 0.15,
   136→          size: 0.5 + Math.random() * 1,
   137→          alpha: 0.05 + Math.random() * 0.1,
   138→        });
   139→      }
   140→    }
   141→
   142→    // Initialize neural mesh nodes (12 nodes distributed inside triangle)
   143→    const initNeuralMesh = (corePositions: { x: number; y: number }[]) => {
   144→      const nodes: NeuralMeshNode[] = [];
   145→      // Distribute 12 nodes inside the triangle using barycentric coordinates
   146→      for (let i = 0; i < 12; i++) {
   147→        const t1 = 0.15 + Math.random() * 0.7;
   148→        const t2 = 0.15 + Math.random() * (1 - t1 - 0.15);
   149→        const t3 = Math.max(0.15, 1 - t1 - t2);
   150→        const total = t1 + t2 + t3;
   151→        const nx = (corePositions[0].x * t1 + corePositions[1].x * t2 + corePositions[2].x * t3) / total;
   152→        const ny = (corePositions[0].y * t1 + corePositions[1].y * t2 + corePositions[2].y * t3) / total;
   153→        nodes.push({ x: nx, y: ny, connections: [], activation: 0 });
   154→      }
   155→      // Connect each node to 3-4 nearest neighbors
   156→      nodes.forEach((node, i) => {
   157→        const distances = nodes.map((n, j) => ({
   158→          idx: j,
   159→          dist: Math.hypot(n.x - node.x, n.y - node.y),
   160→        })).filter(d => d.idx !== i).sort((a, b) => a.dist - b.dist);
   161→        const numConnections = 3 + (i % 2); // 3 or 4
   162→        node.connections = distances.slice(0, numConnections).map(d => d.idx);
   163→      });
   164→      return nodes;
   165→    };
   166→
   167→    // Initialize coherence arcs for each core
   168→    const initCoherenceArcs = (): CoherenceArc[] => {
   169→      return CORES.map((core, i) => ({
   170→        startAngle: (Math.PI * 2 / 3) * i,
   171→        endAngle: (Math.PI * 2 / 3) * i + Math.PI * 0.4,
   172→        radius: 48 + i * 6,
   173→        coherence: 0,
   174→        color: core.color,
   175→      }));
   176→    };
   177→
   178→    let meshInitialized = false;
   179→
   180→    const getCorePos = (coreIdx: number, w: number, h: number) => {
   181→      const cx = w / 2;
   182→      const cy = h / 2;
   183→      const radius = Math.min(w, h) * 0.28;
   184→      const angle = CORES[coreIdx].angle + time * 0.08;
   185→      return {
   186→        x: cx + Math.cos(angle) * radius,
   187→        y: cy + Math.sin(angle) * radius,
   188→      };
   189→    };
   190→
   191→    const spawnParticle = (w: number, h: number) => {
   192→      const source = Math.floor(Math.random() * 3);
   193→      let target = Math.floor(Math.random() * 3);
   194→      while (target === source) target = Math.floor(Math.random() * 3);
   195→      const sp = getCorePos(source, w, h);
   196→      const colors = ['#06d6a0', '#fbbf24', '#e040a0'];
   197→      particlesRef.current.push({
   198→        x: sp.x,
   199→        y: sp.y,
   200→        sourceCore: source,
   201→        targetCore: target,
   202→        progress: 0,
   203→        speed: 0.008 + Math.random() * 0.012,
   204→        color: colors[source],
   205→        size: 2 + Math.random() * 2.5,
   206→        trail: [],
   207→      });
   208→    };
   209→
   210→    // Draw hexagonal grid
   211→    const drawHexGrid = (w: number, h: number) => {
   212→      const hexSize = 30;
   213→      const hexH = hexSize * Math.sqrt(3);
   214→      const hexW = hexSize * 2;
   215→
   216→      ctx.strokeStyle = 'rgba(168, 85, 247, 0.025)';
   217→      ctx.lineWidth = 0.5;
   218→
   219→      for (let row = -1; row < h / hexH + 1; row++) {
   220→        for (let col = -1; col < w / hexW + 1; col++) {
   221→          const offsetX = row % 2 !== 0 ? hexW * 0.75 : 0;
   222→          const cx = col * hexW * 1.5 + offsetX;
   223→          const cy = row * hexH;
   224→
   225→          ctx.beginPath();
   226→          for (let i = 0; i < 6; i++) {
   227→            const angle = (Math.PI / 3) * i - Math.PI / 6;
   228→            const hx = cx + hexSize * Math.cos(angle);
   229→            const hy = cy + hexSize * Math.sin(angle);
   230→            if (i === 0) ctx.moveTo(hx, hy);
   231→            else ctx.lineTo(hx, hy);
   232→          }
   233→          ctx.closePath();
   234→          ctx.stroke();
   235→        }
   236→      }
   237→    };
   238→
   239→    // Draw ambient particles
   240→    const drawAmbientParticles = (w: number, h: number) => {
   241→      ambientParticlesRef.current.forEach(p => {
   242→        p.x += p.vx;
   243→        p.y += p.vy;
   244→        if (p.x < -20) p.x = w + 20;
   245→        if (p.x > w + 20) p.x = -20;
   246→        if (p.y < -20) p.y = h + 20;
   247→        if (p.y > h + 20) p.y = -20;
   248→
   249→        ctx.beginPath();
   250→        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
   251→        ctx.fillStyle = `rgba(168, 85, 247, ${p.alpha})`;
   252→        ctx.fill();
   253→      });
   254→    };
   255→
   256→    // Draw orbital data streams
   257→    const drawOrbitalStreams = (cx: number, cy: number, w: number, h: number) => {
   258→      if (!isActive) return;
   259→      const baseRadius = Math.min(w, h) * 0.18;
   260→      const streamColors = ['#06d6a0', '#fbbf24', '#e040a0'];
   261→      const streamAlphas = [0.12, 0.1, 0.12];
   262→
   263→      for (let s = 0; s < 3; s++) {
   264→        const rx = baseRadius + s * 25;
   265→        const ry = baseRadius * 0.45 + s * 12;
   266→        const rotation = s * 0.4 + time * 0.3 * (s % 2 === 0 ? 1 : -1);
   267→
   268→        ctx.save();
   269→        ctx.translate(cx, cy);
   270→        ctx.rotate(rotation);
   271→
   272→        // Draw elliptical path
   273→        ctx.beginPath();
   274→        ctx.ellipse(0, 0, rx, ry, 0, 0, Math.PI * 2);
   275→        ctx.strokeStyle = streamColors[s];
   276→        ctx.globalAlpha = streamAlphas[s] * syncIntensity;
   277→        ctx.lineWidth = 0.8;
   278→        ctx.setLineDash([4, 8]);
   279→        ctx.lineDashOffset = -time * 60 * (s % 2 === 0 ? 1 : -1);
   280→        ctx.stroke();
   281→        ctx.setLineDash([]);
   282→        ctx.globalAlpha = 1;
   283→
   284→        // Flowing dots along the path
   285→        const numDots = 3 + s;
   286→        for (let d = 0; d < numDots; d++) {
   287→          const dotAngle = (Math.PI * 2 / numDots) * d + time * (1.2 + s * 0.3) * (s % 2 === 0 ? 1 : -1);
   288→          const dx = Math.cos(dotAngle) * rx;
   289→          const dy = Math.sin(dotAngle) * ry;
   290→
   291→          ctx.beginPath();
   292→          ctx.arc(dx, dy, 1.5 + Math.sin(time * 3 + d) * 0.5, 0, Math.PI * 2);
   293→          ctx.fillStyle = streamColors[s];
   294→          ctx.globalAlpha = 0.6 * syncIntensity;
   295→          ctx.fill();
   296→
   297→          // Glow
   298→          ctx.beginPath();
   299→          ctx.arc(dx, dy, 5, 0, Math.PI * 2);
   300→          ctx.fillStyle = streamColors[s];
   301→          ctx.globalAlpha = 0.1 * syncIntensity;
   302→          ctx.fill();
   303→          ctx.globalAlpha = 1;
   304→        }
   305→
   306→        ctx.restore();
   307→      }
   308→    };
   309→
   310→    // Draw pulse waves
   311→    const updateAndDrawPulseWaves = () => {
   312→      pulseWavesRef.current = pulseWavesRef.current.filter(pw => pw.alpha > 0.01);
   313→      pulseWavesRef.current.forEach(pw => {
   314→        pw.radius += 2.5;
   315→        pw.alpha *= 0.96;
   316→
   317→        ctx.beginPath();
   318→        ctx.arc(pw.x, pw.y, pw.radius, 0, Math.PI * 2);
   319→        ctx.strokeStyle = pw.color;
   320→        ctx.globalAlpha = pw.alpha;
   321→        ctx.lineWidth = 2;
   322→        ctx.stroke();
   323→        ctx.globalAlpha = 1;
   324→      });
   325→    };
   326→
   327→    // Check for power changes and emit pulse waves
   328→    const checkPowerChanges = (corePositions: { x: number; y: number }[]) => {
   329→      CORES.forEach((core, i) => {
   330→        const currentPower = corePower[core.id] || 0;
   331→        const prevPower = prevPowerRef.current[core.id] || 0;
   332→        if (Math.abs(currentPower - prevPower) > 0.15 && currentPower > 0.1) {
   333→          pulseWavesRef.current.push({
   334→            x: corePositions[i].x,
   335→            y: corePositions[i].y,
   336→            radius: 30,
   337→            maxRadius: 120,
   338→            color: core.color,
   339→            alpha: 0.6,
   340→          });
   341→        }
   342→        prevPowerRef.current[core.id] = currentPower;
   343→      });
   344→    };
   345→
   346→    // Draw core satellites
   347→    const drawCoreSatellites = (pos: { x: number; y: number }, coreIdx: number, core: CoreDef) => {
   348→      const sats = coreSatellitesRef.current[coreIdx];
   349→      sats.forEach((sat, si) => {
   350→        sat.angle += sat.speed * 0.01;
   351→        const sx = pos.x + Math.cos(sat.angle) * sat.distance;
   352→        const sy = pos.y + Math.sin(sat.angle) * sat.distance;
   353→
   354→        ctx.beginPath();
   355→        ctx.arc(sx, sy, 1.5 + si * 0.3, 0, Math.PI * 2);
   356→        ctx.fillStyle = core.color;
   357→        ctx.globalAlpha = 0.7;
   358→        ctx.fill();
   359→
   360→        // Satellite orbit line
   361→        ctx.beginPath();
   362→        ctx.arc(pos.x, pos.y, sat.distance, sat.angle, sat.angle + 0.8);
   363→        ctx.strokeStyle = core.color;
   364→        ctx.globalAlpha = 0.1;
   365→        ctx.lineWidth = 0.5;
   366→        ctx.stroke();
   367→        ctx.globalAlpha = 1;
   368→      });
   369→    };
   370→
   371→    // Draw tiny bar chart inside core
   372→    const drawCoreBarChart = (pos: { x: number; y: number }, coreIdx: number, core: CoreDef) => {
   373→      const throughput = coreThroughputRef.current[coreIdx];
   374→      const barW = 3;
   375→      const barGap = 1;
   376→      const totalW = throughput.values.length * (barW + barGap);
   377→      const startX = pos.x - totalW / 2;
   378→      const baseY = pos.y + 8;
   379→      const maxBarH = 10;
   380→
   381→      throughput.values.forEach((val, i) => {
   382→        const barH = Math.max(1, (val / 80) * maxBarH);
   383→        const x = startX + i * (barW + barGap);
   384→
   385→        ctx.fillStyle = core.color;
   386→        ctx.globalAlpha = 0.3;
   387→        ctx.fillRect(x, baseY - barH, barW, barH);
   388→        ctx.globalAlpha = 1;
   389→      });
   390→    };
   391→
   392→    // Draw neural network mesh between cores
   393→    const drawNeuralMesh = (corePositions: { x: number; y: number }[]) => {
   394→      if (!isActive) return;
   395→
   396→      // Initialize mesh nodes once based on current core positions (re-init periodically)
   397→      if (!meshInitialized || neuralMeshRef.current.length === 0) {
   398→        neuralMeshRef.current = initNeuralMesh(corePositions);
   399→        meshInitialized = true;
   400→      }
   401→
   402→      // Update node positions to follow the triangle (barycentric interpolation)
   403→      const positions = neuralMeshRef.current.map((node, idx) => {
   404→        // Keep nodes roughly distributed inside the moving triangle
   405→        const ratio1 = 0.15 + (idx % 4) * 0.2;
   406→        const ratio2 = 0.15 + (idx % 3) * 0.2;
   407→        const ratio3 = Math.max(0.15, 1 - ratio1 - ratio2);
   408→        const total = ratio1 + ratio2 + ratio3;
   409→        const nx = (corePositions[0].x * ratio1 + corePositions[1].x * ratio2 + corePositions[2].x * ratio3) / total;
   410→        const ny = (corePositions[0].y * ratio1 + corePositions[1].y * ratio2 + corePositions[2].y * ratio3) / total;
   411→        return { x: nx, y: ny };
   412→      });
   413→
   414→      // Update positions and calculate activations
   415→      const avgPower = (corePower['ollama'] || 0 + corePower['llama4'] || 0 + corePower['openai'] || 0) / 3;
   416→      neuralMeshRef.current.forEach((node, i) => {
   417→        node.x = positions[i].x;
   418→        node.y = positions[i].y;
   419→        // Activation pulses based on core power + time-based oscillation
   420→        const coreInfluence = (corePower[CORES[0].id] || 0) * 0.3 + (corePower[CORES[1].id] || 0) * 0.3 + (corePower[CORES[2].id] || 0) * 0.4;
   421→        node.activation = coreInfluence * (0.5 + 0.5 * Math.sin(time * 3 + i * 0.8));
   422→      });
   423→
   424→      // Draw connections (faint lines)
   425→      const drawnPairs = new Set<string>();
   426→      neuralMeshRef.current.forEach((node, i) => {
   427→        node.connections.forEach(j => {
   428→          const pairKey = `${Math.min(i, j)}-${Math.max(i, j)}`;
   429→          if (drawnPairs.has(pairKey)) return;
   430→          drawnPairs.add(pairKey);
   431→          const target = neuralMeshRef.current[j];
   432→          if (!target) return;
   433→
   434→          const avgActivation = (node.activation + target.activation) / 2;
   435→          ctx.beginPath();
   436→          ctx.moveTo(node.x, node.y);
   437→          ctx.lineTo(target.x, target.y);
   438→          ctx.strokeStyle = `rgba(168, 85, 247, ${0.04 + avgActivation * 0.12})`;
   439→          ctx.lineWidth = 0.5 + avgActivation * 0.5;
   440→          ctx.stroke();
   441→        });
   442→      });
   443→
   444→      // Draw mesh nodes (small circles with brightness = activation)
   445→      neuralMeshRef.current.forEach((node) => {
   446→        const nodeRadius = 2 + node.activation * 2;
   447→        const nodeAlpha = 0.15 + node.activation * 0.5;
   448→
   449→        // Glow
   450→        ctx.beginPath();
   451→        ctx.arc(node.x, node.y, nodeRadius + 4, 0, Math.PI * 2);
   452→        ctx.fillStyle = `rgba(168, 85, 247, ${nodeAlpha * 0.2})`;
   453→        ctx.fill();
   454→
   455→        // Core dot
   456→        ctx.beginPath();
   457→        ctx.arc(node.x, node.y, nodeRadius, 0, Math.PI * 2);
   458→        const nodeGrad = ctx.createRadialGradient(node.x, node.y, 0, node.x, node.y, nodeRadius);
   459→        nodeGrad.addColorStop(0, `rgba(200, 170, 255, ${nodeAlpha})`);
   460→        nodeGrad.addColorStop(1, `rgba(168, 85, 247, ${nodeAlpha * 0.3})`);
   461→        ctx.fillStyle = nodeGrad;
   462→        ctx.fill();
   463→      });
   464→    };
   465→
   466→    // Draw interference patterns between cores
   467→    const drawInterferencePatterns = (corePositions: { x: number; y: number }[]) => {
   468→      if (!isActive || syncIntensity < 0.5) return;
   469→
   470→      const interferenceAlpha = (syncIntensity - 0.5) * 2; // 0 to 1 as sync goes 0.5 to 1
   471→      const ringSpacing = 18;
   472→      const maxRings = 8;
   473→
   474→      for (let c = 0; c < 3; c++) {
   475→        const pos = corePositions[c];
   476→        for (let r = 1; r <= maxRings; r++) {
   477→          const radius = r * ringSpacing + Math.sin(time * 2 + r) * 3;
   478→          const ringAlpha = (0.02 + interferenceAlpha * 0.02) * (1 - r / (maxRings + 1));
   479→          ctx.beginPath();
   480→          ctx.arc(pos.x, pos.y, radius, 0, Math.PI * 2);
   481→          ctx.strokeStyle = CORES[c].color;
   482→          ctx.globalAlpha = ringAlpha;
   483→          ctx.lineWidth = 1.5;
   484→          ctx.stroke();
   485→          ctx.globalAlpha = 1;
   486→        }
   487→      }
   488→    };
   489→
   490→    // Draw quantum coherence arcs around nexus
   491→    const drawCoherenceArcs = (cx: number, cy: number) => {
   492→      if (!isActive) return;
   493→
   494→      if (coherenceArcsRef.current.length === 0) {
   495→        coherenceArcsRef.current = initCoherenceArcs();
   496→      }
   497→
   498→      coherenceArcsRef.current.forEach((arc, i) => {
   499→        const power = corePower[CORES[i].id] || 0;
   500→        arc.coherence = power;
   501→
   502→        // Arc length proportional to core's power/coherence
   503→        const arcLength = Math.PI * 0.4 * arc.coherence;
   504→        const startAngle = (Math.PI * 2 / 3) * i + time * 0.2;
   505→        const endAngle = startAngle + arcLength;
   506→        const radius = 48 + i * 6;
   507→
   508→        if (arcLength < 0.05) return;
   509→
   510→        // Draw arc
   511→        ctx.beginPath();
   512→        ctx.arc(cx, cy, radius, startAngle, endAngle);
   513→        ctx.strokeStyle = arc.color;
   514→        ctx.globalAlpha = 0.35 * arc.coherence;
   515→        ctx.lineWidth = 2.5;
   516→        ctx.lineCap = 'round';
   517→        ctx.stroke();
   518→        ctx.lineCap = 'butt';
   519→        ctx.globalAlpha = 1;
   520→
   521→        // Pulsing glow at arc start endpoint
   522→        const glowPulse = 0.5 + 0.5 * Math.sin(time * 4 + i * 2);
   523→        const startPx = cx + Math.cos(startAngle) * radius;
   524→        const startPy = cy + Math.sin(startAngle) * radius;
   525→        ctx.beginPath();
   526→        ctx.arc(startPx, startPy, 4 + glowPulse * 2, 0, Math.PI * 2);
