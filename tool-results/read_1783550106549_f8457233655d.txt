    65тЖТ    65тЖТ  useEffect(() => {

    69тЖТ    69тЖТ    if (!ctx) return;
    70тЖТ    70тЖТ
    71тЖТ    71тЖТ    const rect = canvas.getBoundingClientRect();
    72тЖТ    72тЖТ    const dpr = window.devicePixelRatio || 1;

    74тЖТ    74тЖТ    canvas.height = rect.height * dpr;
    75тЖТ    75тЖТ    ctx.scale(dpr, dpr);
    76тЖТ    76тЖТ

    83тЖТ    83тЖТ      const base = 75 + cycleCount * 2;

   100тЖТ   100тЖТ
   101тЖТ   101тЖТ    ctx.beginPath();
   102тЖТ   102тЖТ    data.forEach((val, i) => {
   103тЖТ   103тЖТ      const x = padX + (i / (data.length - 1)) * chartW;
   104тЖТ   104тЖТ      const y = padY + chartH - ((val - minVal) / range) * chartH;
   105тЖТ   105тЖТ      if (i === 0) ctx.moveTo(x, y);
   106тЖТ   106тЖТ      else ctx.lineTo(x, y);
   107тЖТ   107тЖТ    });
   108тЖТ   108тЖТ
   109тЖТ   109тЖТ    // Close path for fill
   110тЖТ   110тЖТ    ctx.lineTo(padX + chartW, padY + chartH);
   111тЖТ   111тЖТ    ctx.lineTo(padX, padY + chartH);
   112тЖТ   112тЖТ    ctx.closePath();
   113тЖТ   113тЖТ    ctx.fillStyle = grad;
   114тЖТ   114тЖТ    ctx.fill();
   115тЖТ   115тЖТ
   116тЖТ   116тЖТ    // Line
   117тЖТ   117тЖТ    ctx.beginPath();
   118тЖТ   118тЖТ    data.forEach((val, i) => {
   119тЖТ   119тЖТ      const x = padX + (i / (data.length - 1)) * chartW;
   120тЖТ   120тЖТ      const y = padY + chartH - ((val - minVal) / range) * chartH;
   121тЖТ   121тЖТ      if (i === 0) ctx.moveTo(x, y);
   122тЖТ   122тЖТ      else ctx.lineTo(x, y);
   123тЖТ   123тЖТ    });
   124тЖТ   124тЖТ    ctx.strokeStyle = '#a855f7';
   125тЖТ   125тЖТ    ctx.lineWidth = 1.5;
   126тЖТ   126тЖТ    ctx.stroke();
   127тЖТ   127тЖТ
   128тЖТ   128тЖТ    // Last point dot
   129тЖТ   129тЖТ    const lastX = padX + chartW;
   130тЖТ   130тЖТ    const lastY = padY + chartH - ((data[data.length - 1] - minVal) / range) * chartH;
   131тЖТ   131тЖТ    ctx.beginPath();
   132тЖТ   132тЖТ    ctx.arc(lastX, lastY, 2.5, 0, Math.PI * 2);
   133тЖТ   133тЖТ    ctx.fillStyle = '#a855f7';
   134тЖТ   134тЖТ    ctx.fill();
   135тЖТ   135тЖТ    ctx.beginPath();
   136тЖТ   136тЖТ    ctx.arc(lastX, lastY, 5, 0, Math.PI * 2);
   137тЖТ   137тЖТ    ctx.fillStyle = 'rgba(168, 85, 247, 0.25)';
   138тЖТ   138тЖТ    ctx.fill();
   139тЖТ   139тЖТ  }, [cycleCount]);
   140тЖТ   140тЖТ
   141тЖТ   141тЖТ  return (
   142тЖТ   142тЖТ    <div className="flex items-center gap-2">
   143тЖТ   143тЖТ      <span className="text-[9px] font-mono text-[#8888aa] uppercase tracking-wider">RAG Perf</span>

   148тЖТ   148тЖТ      />
   149тЖТ   149тЖТ    </div>
