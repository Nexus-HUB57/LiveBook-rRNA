   119→/* ---------- Explorer Tab ---------- */
   120→function ExplorerTab() {
   121→  const [projects, setProjects] = useState<Project[]>([]);
   122→  const [total, setTotal] = useState(2402);
   123→  const [categories, setCategories] = useState<string[]>([]);
   124→  const [sources, setSources] = useState<string[]>([]);
   125→  const [search, setSearch] = useState('');
   126→  const [activeCat, setActiveCat] = useState('');
   127→  const [source, setSource] = useState('');
   128→  const [sort, setSort] = useState('newest');
   129→  const [page, setPage] = useState(1);
   130→  const [loading, setLoading] = useState(true);
   131→  const limit = 20;
   132→  const totalPages = Math.ceil(total / limit);
   133→
   134→  const fetchProjects = useCallback(async (p: number) => {
   135→    setLoading(true);
   136→    try {
   137→      const params = new URLSearchParams({ page: String(p), limit: String(limit) });
   138→      if (search) params.set('search', search);
   139→      if (activeCat) params.set('category', activeCat);
   140→      if (source) params.set('source', source);
   141→      params.set('sort', sort);
   142→      const res = await fetch(`/api/projects?${params}`);
   143→      const data = await res.json();
   144→      setProjects(data.projects || []);
   145→      setTotal(data.total || 2402);
   146→      if (data.categories) setCategories(data.categories);
   147→      if (data.sources) setSources(data.sources);
   148→    } catch { setProjects([]); }
   149→    setLoading(false);
   150→  }, [search, activeCat, source, sort]);
   151→
   152→  // eslint-disable-next-line react-hooks/set-state-in-effect
   153→  useEffect(() => { fetchProjects(page); }, [fetchProjects, page]);
   154→
   155→  return (
   156→    <div className="space-y-4">
   157→      <div className="relative">
   158→        <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-zinc-500" />
   159→        <Input placeholder="Search 2,402 projects..." value={search}
   160→          onChange={e => { setSearch(e.target.value); setPage(1); }}
   161→          className="pl-10 bg-zinc-900 border-zinc-800 text-zinc-100 placeholder:text-zinc-600 h-11 rounded-xl" />
   162→      </div>
   163→      <div className="flex flex-wrap items-center gap-2">
   164→        <Filter className="w-4 h-4 text-zinc-500" />
   165→        <ScrollArea className="w-full">
   166→          <div className="flex gap-1.5 pb-1">
   167→            <button onClick={() => { setActiveCat(''); setPage(1); }}
   168→              className={`px-3 py-1 rounded-full text-xs font-medium transition-all border ${!activeCat ? 'bg-emerald-500/20 text-emerald-400 border-emerald-500/30' : 'border-zinc-800 text-zinc-500 hover:text-zinc-300 hover:border-zinc-700'}`}>
   169→              All
   170→            </button>
   171→            {categories.map(c => (
   172→              <button key={c} onClick={() => { setActiveCat(c === activeCat ? '' : c); setPage(1); }}
   173→                className={`px-3 py-1 rounded-full text-xs font-medium transition-all border ${c === activeCat ? catColor(c) : 'border-zinc-800 text-zinc-500 hover:text-zinc-300 hover:border-zinc-700'}`}>
   174→                {c}
   175→              </button>
   176→            ))}
   177→          </div>
   178→        </ScrollArea>
   179→      </div>
   180→      <div className="flex flex-wrap items-center gap-3">
   181→        <Select value={source} onValueChange={v => { setSource(v === 'all' ? '' : v); setPage(1); }}>
   182→          <SelectTrigger className="w-[140px] bg-zinc-900 border-zinc-800 text-zinc-300 h-8 text-xs">
   183→            <SelectValue placeholder="Source" />
   184→          </SelectTrigger>
   185→          <SelectContent className="bg-zinc-900 border-zinc-800">
   186→            <SelectItem value="all" className="text-zinc-300">All Sources</SelectItem>
   187→            {sources.map(s => <SelectItem key={s} value={s} className="text-zinc-300">{s}</SelectItem>)}
   188→          </SelectContent>
   189→        </Select>
   190→        <Select value={sort} onValueChange={v => setSort(v)}>
   191→          <SelectTrigger className="w-[130px] bg-zinc-900 border-zinc-800 text-zinc-300 h-8 text-xs">
   192→            <SelectValue />
   193→          </SelectTrigger>
   194→          <SelectContent className="bg-zinc-900 border-zinc-800">
   195→            <SelectItem value="newest" className="text-zinc-300">Newest</SelectItem>
   196→            <SelectItem value="oldest" className="text-zinc-300">Oldest</SelectItem>
   197→          </SelectContent>
   198→        </Select>
   199→        <span className="text-xs text-zinc-600 ml-auto">
   200→          {total.toLocaleString()} projects &bull; {categories.length} categories &bull; 870 dates
   201→        </span>
   202→      </div>
   203→      <Separator className="bg-zinc-800" />
   204→      {loading ? (
   205→        <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-3">
   206→          {Array.from({ length: 6 }).map((_, i) => <ProjectSkeleton key={i} />)}
   207→        </div>
   208→      ) : projects.length === 0 ? (
   209→        <div className="text-center py-20">
   210→          <Search className="w-10 h-10 text-zinc-700 mx-auto mb-3" />
   211→          <p className="text-zinc-500 text-sm">No projects found</p>
   212→        </div>
   213→      ) : (
   214→        <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 gap-3">
   215→          {projects.map(p => <ProjectCard key={p.id} p={p} />)}
   216→        </div>
   217→      )}
   218→      <div className="flex items-center justify-center gap-2 pt-2">
   219→        <Button variant="outline" size="sm" disabled={page <= 1} onClick={() => setPage(p => p - 1)}
   220→          className="border-zinc-800 text-zinc-400 hover:text-zinc-100 hover:bg-zinc-800">
   221→          <ChevronLeft className="w-4 h-4" />Prev
   222→        </Button>
   223→        <span className="text-xs text-zinc-500 px-3">Page {page} of {totalPages}</span>
   224→        <Button variant="outline" size="sm" disabled={page >= totalPages} onClick={() => setPage(p => p + 1)}
   225→          className="border-zinc-800 text-zinc-400 hover:text-zinc-100 hover:bg-zinc-800">
   226→          Next<ChevronRight className="w-4 h-4" />
   227→        </Button>
   228→        {page < totalPages && (
   229→          <Button variant="ghost" size="sm" onClick={() => setPage(p => p + 1)}
   230→            className="text-emerald-400 hover:text-emerald-300 hover:bg-emerald-500/10 ml-2">
   231→            Load more...
   232→          </Button>
   233→        )}
   234→      </div>
   235→    </div>
   236→  );
   237→}

   239→/* ---------- Chat Tab ---------- */
   240→function ChatTab() {
   241→  const [messages, setMessages] = useState<ChatMsg[]>([
   242→    { role: 'system', content: "I'm your AI Agent. Ask me about the 2402+ projects, get recommendations, or analyze trends." }
   243→  ]);
   244→  const [input, setInput] = useState('');
   245→  const [loading, setLoading] = useState(false);
   246→  const scrollRef = useRef<HTMLDivElement>(null);
   247→
   248→  useEffect(() => {
   249→    if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
   250→  }, [messages]);
   251→
   252→  const send = useCallback(async (text?: string) => {
   253→    const msg = text || input.trim();
   254→    if (!msg || loading) return;
   255→    setMessages(m => [...m, { role: 'user', content: msg }]);
   256→    setInput('');
   257→    setLoading(true);
   258→    try {
   259→      const res = await fetch('/api/agent/chat?XTransformPort=3001', {
   260→        method: 'POST', headers: { 'Content-Type': 'application/json' },
   261→        body: JSON.stringify({
   262→          messages: [...messages, { role: 'user', content: msg }],
   263→          context: 'You are an expert on Chinese independent developer projects. Help users explore, discover, and analyze the 2402+ projects in this ecosystem.'
   264→        })
   265→      });
   266→      const data = await res.json();
   267→      setMessages(m => [...m, { role: 'agent', content: data.response || 'No response received.' }]);
   268→    } catch {
   269→      setMessages(m => [...m, { role: 'agent', content: 'Sorry, the AI agent is currently unavailable. Please try again.' }]);
   270→    }
   271→    setLoading(false);
   272→  }, [input, messages, loading]);
   273→
   274→  const quickActions = ['Recommend projects', 'Analyze trends', 'Top categories'];
   275→
   276→  return (
   277→    <div className="flex flex-col h-[70vh] max-h-[80vh]">
   278→      <ScrollArea className="flex-1 mb-4" ref={scrollRef as React.RefObject<any>}>
   279→        <div className="space-y-3 pr-4">
   280→          {messages.map((m, i) => (
   281→            <div key={i} className={`flex ${m.role === 'user' ? 'justify-end' : 'justify-start'}`}>
   282→              <div className={`max-w-[85%] md:max-w-[70%] rounded-2xl px-4 py-2.5 text-sm leading-relaxed ${
   283→                m.role === 'user'
   284→                  ? 'bg-emerald-600/90 text-white rounded-br-md'
   285→                  : m.role === 'system'
   286→                  ? 'bg-zinc-800/80 text-zinc-300 text-center w-full max-w-full border border-zinc-700/50 rounded-2xl'
   287→                  : 'bg-zinc-800 text-zinc-200 rounded-bl-md'
   288→              }`}>
   289→                {m.role === 'agent' && (
   290→                  <div className="flex items-center gap-1.5 mb-1.5 text-emerald-400 text-[10px] font-medium">
   291→                    <Bot className="w-3 h-3" />AI Agent
   292→                  </div>
   293→                )}
   294→                {m.content}
   295→              </div>
   296→            </div>
   297→          ))}
   298→          {loading && (
   299→            <div className="flex justify-start">
   300→              <div className="bg-zinc-800 rounded-2xl rounded-bl-md px-4 py-3 space-y-2">
   301→                <Skeleton className="h-3 w-48 bg-zinc-700" />
   302→                <Skeleton className="h-3 w-36 bg-zinc-700" />
   303→                <Skeleton className="h-3 w-24 bg-zinc-700" />
   304→              </div>
   305→            </div>
   306→          )}
   307→        </div>
   308→      </ScrollArea>
   309→      <Separator className="bg-zinc-800 mb-3" />
   310→      <div className="flex gap-1.5 mb-2 flex-wrap">
   311→        {quickActions.map(a => (
   312→          <Button key={a} variant="outline" size="sm"
   313→            onClick={() => send(a)}
   314→            className="border-zinc-800 text-zinc-400 hover:text-emerald-400 hover:border-emerald-500/30 hover:bg-emerald-500/5 text-xs h-7">
   315→            <Sparkles className="w-3 h-3" />{a}
   316→          </Button>
   317→        ))}
   318→      </div>
   319→      <div className="flex gap-2">
   320→        <Input value={input} onChange={e => setInput(e.target.value)}
   321→          onKeyDown={e => e.key === 'Enter' && !e.shiftKey && send()}
   322→          placeholder="Ask about projects, trends, recommendations..."
   323→          className="flex-1 bg-zinc-900 border-zinc-800 text-zinc-100 placeholder:text-zinc-600 rounded-xl" />
   324→        <Button onClick={() => send()} disabled={loading || !input.trim()}
   325→          className="bg-emerald-600 hover:bg-emerald-700 text-white rounded-xl px-5">
   326→          {loading ? <Loader2 className="w-4 h-4 animate-spin" /> : <Send className="w-4 h-4" />}
   327→        </Button>
   328→      </div>
   329→    </div>
   330→  );
   331→}

   428→  const [stats, setStats] = useState<Stats | null>(null);
   429→  const [loading, setLoading] = useState(true);
   430→  const [insight, setInsight] = useState('');
   431→  const [insightLoading, setInsightLoading] = useState(false);
   432→
   433→  useEffect(() => {
   434→    (async () => {
   435→      try {
   436→        const res = await fetch('/api/projects/stats');
   437→        const data = await res.json();
   438→        setStats(data);
   439→      } catch { setStats(null); }
   440→      setLoading(false);
   441→    })();
   442→  }, []);
   443→
   444→  const analyze = async () => {
   445→    setInsightLoading(true);
   446→    try {
   447→      const res = await fetch('/api/agent/analyze?XTransformPort=3001', {
   448→        method: 'POST', headers: { 'Content-Type': 'application/json' },
   449→        body: JSON.stringify({ query: 'Analyze the top trends and insights from the Chinese independent developer project ecosystem' })
   450→      });
   451→      const data = await res.json();
   452→      setInsight(data.analysis || 'No analysis available');
   453→    } catch { setInsight('Analysis service unavailable.'); }
   454→    setInsightLoading(false);
   455→  };
   456→
   457→  const catEntries = stats?.byCategory ? Object.entries(stats.byCategory).sort((a, b) => b[1] - a[1]) : [];
   458→  const maxCat = catEntries[0]?.[1] || 1;
   459→  const srcEntries = stats?.bySource ? Object.entries(stats.bySource) : [];
   460→  const monthData = stats?.byMonth || [];
   461→  const maxMonth = monthData.length > 0 ? Math.max(...monthData.map(m => m.count)) : 1;
   462→
   463→  const PIE_COLORS = ['bg-emerald-500', 'bg-amber-500', 'bg-purple-500', 'bg-sky-500', 'bg-pink-500', 'bg-teal-500'];
   464→
   465→  if (loading) {
   466→    return (
   467→      <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
   468→        {Array.from({ length: 4 }).map((_, i) => (
   469→          <Skeleton key={i} className="h-24 bg-zinc-900 rounded-xl" />
   470→        ))}
   471→      </div>
   472→    );
   473→  }
   474→
   475→  return (
   476→    <div className="space-y-6">
   477→      {/* Stat Cards */}
   478→      <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
   479→        {[
   480→          { label: 'Total Projects', value: stats?.total?.toLocaleString() || '2,402', icon: Globe, color: 'text-emerald-400' },
   481→          { label: 'Active', value: String(stats?.total ? Math.round(stats.total * 0.72) : 1730), icon: Zap, color: 'text-amber-400' },
   482→          { label: 'Categories', value: String(catEntries.length), icon: Filter, color: 'text-purple-400' },
   483→          { label: 'Sources', value: String(srcEntries.length), icon: Github, color: 'text-sky-400' },
   484→        ].map(s => (
   485→          <Card key={s.label} className="bg-zinc-900 border-zinc-800 gap-2 py-4 px-4">
   486→            <div className="flex items-center justify-between">
   487→              <span className="text-[10px] text-zinc-500 uppercase tracking-wider">{s.label}</span>
   488→              <s.icon className={`w-4 h-4 ${s.color}`} />
   489→            </div>
   490→            <div className="text-2xl font-bold text-zinc-100">{s.value}</div>
   491→          </Card>
   492→        ))}
   493→      </div>
   494→
   495→      <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
   496→        {/* Category Breakdown */}
   497→        <Card className="bg-zinc-900 border-zinc-800 gap-3">
   498→          <CardHeader className="pb-0">
   499→            <CardTitle className="text-sm text-zinc-200 flex items-center gap-2">
   500→              <Filter className="w-4 h-4 text-emerald-400" />Category Breakdown
   501→            </CardTitle>
   502→          </CardHeader>
   503→          <CardContent className="space-y-2">
   504→            <ScrollArea className="max-h-64">
   505→              {catEntries.map(([cat, count]) => (
   506→                <div key={cat} className="space-y-1">
   507→                  <div className="flex justify-between text-xs">
   508→                    <span className="text-zinc-400">{cat}</span>
   509→                    <span className="text-zinc-500">{count}</span>
   510→                  </div>
   511→                  <div className="h-2 bg-zinc-800 rounded-full overflow-hidden">
   512→                    <div className={`h-full rounded-full transition-all duration-500 ${catColor(cat).split(' ')[0]}`}
   513→                      style={{ width: `${(count / maxCat) * 100}%` }} />
   514→                  </div>
   515→                </div>
   516→              ))}
   517→            </ScrollArea>
   518→          </CardContent>
   519→        </Card>
   520→
   521→        {/* Monthly Trend */}
   522→        <Card className="bg-zinc-900 border-zinc-800 gap-3">
   523→          <CardHeader className="pb-0">
   524→            <CardTitle className="text-sm text-zinc-200 flex items-center gap-2">
   525→              <TrendingUp className="w-4 h-4 text-amber-400" />Monthly Trend
   526→            </CardTitle>
   527→          </CardHeader>
   528→          <CardContent>
   529→            <div className="flex items-end gap-[2px] h-48">
   530→              {monthData.slice(-24).map((m, i) => {
   531→                const h = (m.count / maxMonth) * 100;
   532→                return (
   533→                  <Tooltip key={i}>
   534→                    <TooltipTrigger asChild>
   535→                      <div className="flex-1 min-w-0 flex flex-col items-center justify-end h-full group">
   536→                        <div className="w-full bg-emerald-500/80 hover:bg-emerald-400 rounded-t-sm transition-colors"
   537→                          style={{ height: `${h}%`, minHeight: m.count > 0 ? '2px' : '0' }} />
   538→                      </div>
   539→                    </TooltipTrigger>
   540→                    <TooltipContent className="bg-zinc-800 border-zinc-700 text-zinc-300 text-[10px]">
   541→                      {m.month}: {m.count} projects
   542→                    </TooltipContent>
   543→                  </Tooltip>
   544→                );
   545→              })}
   546→            </div>
   547→            <div className="flex justify-between text-[10px] text-zinc-600 mt-1">
   548→              <span>{monthData[0]?.month || ''}</span>
   549→              <span>{monthData[monthData.length - 1]?.month || ''}</span>
   550→            </div>
   551→          </CardContent>
   552→        </Card>
   553→      </div>
   554→
   555→      <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
   556→        {/* Top Authors */}
   557→        <Card className="bg-zinc-900 border-zinc-800 gap-3">
   558→          <CardHeader className="pb-0">
   559→            <CardTitle className="text-sm text-zinc-200 flex items-center gap-2">
   560→              <Users className="w-4 h-4 text-purple-400" />Top 10 Authors
   561→            </CardTitle>
   562→          </CardHeader>
   563→          <CardContent>
   564→            <ScrollArea className="max-h-64">
   565→              <div className="space-y-1.5">
   566→                {(stats?.topAuthors || []).slice(0, 10).map((a, i) => (
   567→                  <div key={i} className="flex items-center gap-2 text-xs">
   568→                    <span className="w-5 text-right text-zinc-600 font-mono">{i + 1}</span>
   569→                    <div className="flex-1 truncate text-zinc-300">{a.name}</div>
   570→                    <span className="text-zinc-500">{a.count} projects</span>
   571→                  </div>
   572→                ))}
   573→                {(!stats?.topAuthors?.length) && <p className="text-xs text-zinc-600">No author data available</p>}
   574→              </div>
   575→            </ScrollArea>
   576→          </CardContent>
   577→        </Card>
   578→
   579→        {/* Source Distribution */}
   580→        <Card className="bg-zinc-900 border-zinc-800 gap-3">
   581→          <CardHeader className="pb-0">
   582→            <CardTitle className="text-sm text-zinc-200 flex items-center gap-2">
   583→              <Github className="w-4 h-4 text-sky-400" />Source Distribution
   584→            </CardTitle>
   585→          </CardHeader>
   586→          <CardContent>
   587→            {srcEntries.length > 0 ? (
   588→              <div className="flex items-center gap-6">
   589→                <div className="w-28 h-28 rounded-full flex-shrink-0" style={{
   590→                  background: `conic-gradient(${srcEntries.map(([, v], i) => {
   591→                    const pct = (v / srcEntries.reduce((a, [, b]) => a + b, 0)) * 100;
   592→                    const prev = srcEntries.slice(0, i).reduce((a, [, b]) => a + b, 0);
   593→                    const prevPct = (prev / srcEntries.reduce((a, [, b]) => a + b, 0)) * 100;
   594→                    return `${PIE_COLORS[i % PIE_COLORS.length]} ${prevPct}% ${prevPct + pct}%`;
   595→                  }).join(', ')})`
   596→                }} />
   597→                <div className="space-y-1.5 flex-1">
   598→                  {srcEntries.map(([s, v], i) => (
   599→                    <div key={s} className="flex items-center gap-2 text-xs">
   600→                      <span className={`w-2.5 h-2.5 rounded-sm ${PIE_COLORS[i % PIE_COLORS.length]}`} />
   601→                      <span className="text-zinc-400 flex-1">{s}</span>
   602→                      <span className="text-zinc-500">{v}</span>
   603→                    </div>
   604→                  ))}
   605→                </div>
   606→              </div>
   607→            ) : (
   608→              <p className="text-xs text-zinc-600">No source data available</p>
   609→            )}
   610→          </CardContent>
   611→        </Card>
   612→      </div>
   613→
   614→      {/* AI Insights */}
   615→      <Card className="bg-zinc-900 border-zinc-800 gap-3">
   616→        <CardHeader className="pb-0">
   617→          <CardTitle className="text-sm text-zinc-200 flex items-center gap-2">
   618→            <Sparkles className="w-4 h-4 text-emerald-400" />AI Insights
   619→          </CardTitle>
   620→        </CardHeader>
   621→        <CardContent className="space-y-3">
   622→          <Button onClick={analyze} disabled={insightLoading}
   623→            className="bg-emerald-600 hover:bg-emerald-700 text-white">
   624→            {insightLoading ? <Loader2 className="w-4 h-4 animate-spin" /> : <Sparkles className="w-4 h-4" />}
   625→            Generate AI Analysis
   626→          </Button>
   627→          {insight && (
   628→            <div className="p-4 bg-zinc-800/50 rounded-lg border border-zinc-800 text-sm text-zinc-300 whitespace-pre-wrap leading-relaxed">
   629→              {insight}
   630→            </div>
   631→          )}
   632→        </CardContent>
   633→      </Card>
   634→    </div>
   635→  );
   636→}
