    60→function getEmotionalIntensity(node: StoryNode): number {
    61→  const depthBase = [0.15, 0.32, 0.48, 0.62, 0.78, 0.92];
    62→  const eraMod = node.era === 'atemporal' ? 0.1 : node.era === '2077' ? 0.07 : 0.0;
    63→  const hash = parseInt(node.id.slice(1), 10);
    64→  const variation = Math.sin(hash * 2.7) * 0.06;
    65→  return Math.min(1, Math.max(0, depthBase[node.depth] + eraMod + variation));
    66→}
    67→
    68→function polarToCart(cx: number, cy: number, r: number, angleDeg: number) {
    69→  const rad = (angleDeg * Math.PI) / 180;
    70→  return { x: cx + r * Math.cos(rad), y: cy + r * Math.sin(rad) };
    71→}
    72→
    73→function describeArc(cx: number, cy: number, r: number, startDeg: number, endDeg: number): string {
    74→  const start = polarToCart(cx, cy, r, startDeg);
    75→  const end = polarToCart(cx, cy, r, endDeg);
    76→  const sweep = endDeg - startDeg;
    77→  const large = sweep > 180 ? 1 : 0;
    78→  return `M ${start.x} ${start.y} A ${r} ${r} 0 ${large} 1 ${end.x} ${end.y}`;
    79→}
