   103→// Draw a rounded rect using arcTo (no roundRect)
   104→function drawRoundedRect(
   105→  ctx: CanvasRenderingContext2D,
   106→  x: number, y: number, w: number, h: number, r: number
   107→) {
   108→  ctx.moveTo(x + r, y);
   109→  ctx.lineTo(x + w - r, y);
   110→  ctx.arcTo(x + w, y, x + w, y + r, r);
   111→  ctx.lineTo(x + w, y + h - r);
   112→  ctx.arcTo(x + w, y + h, x + w - r, y + h, r);
   113→  ctx.lineTo(x + r, y + h);
   114→  ctx.arcTo(x, y + h, x, y + h - r, r);
   115→  ctx.lineTo(x, y + r);
   116→  ctx.arcTo(x, y, x + r, y, r);
   117→  ctx.closePath();
   118→}

   189→      draggedNodeRef.current = null;
   190→    };

   192→    canvas.addEventListener('mousemove', handleMouseMove);
   193→    canvas.addEventListener('mousedown', handleMouseDown);
   194→    canvas.addEventListener('mouseup', handleMouseUp);

   198→    let hoveredRef: string | null = null;

   200→    // Get unique categories in order

   202→

   207→        const key = [n.id, c].sort().join('::');
   208→        if (!edgeIsCurved.has(key)) {
   209→          edgeIsCurved.set(key, Math.random() > 0.5);

   211→      });

   216→      const w2 = canvas.getBoundingClientRect().width;
   217→      const h2 = canvas.getBoundingClientRect().height;

   219→      const cy = h2 / 2;

   222→      nodes.forEach(n => {
   223→        n.vx += (cx - n.x) * 0.0003;
   224→        n.vy += (cy - n.y) * 0.0003;
   225→      });

   227→      // Repulsion between nodes (reduced for same-cluster)
   228→      for (let i = 0; i < nodes.length; i++) {
   229→        for (let j = i + 1; j < nodes.length; j++) {
   230→          const dx = nodes[j].x - nodes[i].x;
   231→          const dy = nodes[j].y - nodes[i].y;
   232→          const dist = Math.max(1, Math.sqrt(dx * dx + dy * dy));
   233→          const sameCluster = nodes[i].category === nodes[j].category;
   234→          const repulsionRadius = sameCluster ? 100 : 150;
   235→          const repulsionForce = sameCluster ? 0.00015 : 0.0003;
   236→          if (dist < repulsionRadius) {
   237→            const force = (repulsionRadius - dist) * repulsionForce;
   238→            const fx = (dx / dist) * force;
   239→            const fy = (dy / dist) * force;
   240→            nodes[i].vx -= fx;
   241→            nodes[i].vy -= fy;
   242→            nodes[j].vx += fx;
   243→            nodes[j].vy += fy;
   244→          }
   245→        }
   246→      }
   247→
   248→      // Category-based attraction (same category nodes attract more strongly)

   251→          if (nodes[i].category !== nodes[j].category) continue;
   252→          const dx = nodes[j].x - nodes[i].x;
   253→          const dy = nodes[j].y - nodes[i].y;

   258→            nodes[i].vy += (dy / dist) * force;
   259→            nodes[j].vx -= (dx / dist) * force;
   260→            nodes[j].vy -= (dy / dist) * force;
   261→          }
   262→        }
   263→      }
   264→
   265→      // Circular layout bias: nodes tend to arrange by category around a circle
   266→      const categoryAngleMap: Record<string, number> = {};

   270→      const layoutRadius = Math.min(w2, h2) * 0.3;
   271→      nodes.forEach(n => {

   273→        const targetX = cx + Math.cos(targetAngle + time * 0.02) * layoutRadius * 0.4;
   274→        const targetY = cy + Math.sin(targetAngle + time * 0.02) * layoutRadius * 0.4;
   275→        n.vx += (targetX - n.x) * 0.00005;
   276→        n.vy += (targetY - n.y) * 0.00005;

   282→        const dy = n.y - cy;
   283→        const dist = Math.sqrt(dx * dx + dy * dy);
   284→        if (dist > 10) {
   285→          // Tangential force for orbital motion
   286→          const orbitalSpeed = 0.00003;
   287→          n.vx += (-dy / dist) * orbitalSpeed * dist;
   288→          n.vy += (dx / dist) * orbitalSpeed * dist;
   289→        }
   290→      });
   291→
   292→      // Spring connections
   293→      nodes.forEach(n => {
   294→        n.connections.forEach(connId => {
   295→          const target = nodes.find(t => t.id === connId);
   296→          if (!target) return;
   297→          const dx = target.x - n.x;
   298→          const dy = target.y - n.y;
   299→          const dist = Math.max(1, Math.sqrt(dx * dx + dy * dy));
   300→          const idealDist = 100;
   301→          const force = (dist - idealDist) * 0.0008;
   302→          n.vx += (dx / dist) * force;
   303→          n.vy += (dy / dist) * force;
   304→        });
   305→      });
   306→
   307→      // Mouse attraction/repulsion for hovered node
   308→      if (hoveredRef) {
   309→        const node = nodes.find(n => n.id === hoveredRef);
   310→        if (node) {
   311→          const dx = mouseRef.current.x - node.x;
   312→          const dy = mouseRef.current.y - node.y;
   313→          const dist = Math.max(1, Math.sqrt(dx * dx + dy * dy));
   314→          if (dist > 50) {
   315→            node.vx += (dx / dist) * 0.15;
   316→            node.vy += (dy / dist) * 0.15;
   317→          }
   318→        }
   319→      }
   320→
   321→      // Dragging
   322→      if (draggedNodeRef.current) {
   323→        const node = nodes.find(n => n.id === draggedNodeRef.current);

   325→          node.x = mouseRef.current.x;
   326→          node.y = mouseRef.current.y;

   334→          n.vy *= 0.92;

   336→          n.y += n.vy;

   369→      // ---- Draw cluster zones ----

   397→        drawRoundedRect(ctx, zx, zy, zw, zh, zr);

   399→        ctx.fill();
   400→        ctx.strokeStyle = CLUSTER_BORDER_COLORS[category] || 'rgba(136, 136, 170, 0.08)';
   401→        ctx.lineWidth = 1;
   402→        ctx.setLineDash([4, 6]);
   403→        ctx.stroke();
   404→        ctx.setLineDash([]);

   406→        // Category label

   409→        ctx.textAlign = 'left';
   410→        ctx.textBaseline = 'top';
   411→        ctx.fillText(category.toUpperCase(), zx + 8, zy + 5);

   414→      // ---- Quantum pulse wave from active node ----

   417→        if (activeNodeObj) {
   418→          const pulseRadius = ((time * 80) % 200);
   419→          const pulseAlpha = Math.max(0, 1 - pulseRadius / 200) * 0.15;
   420→          if (pulseAlpha > 0.01) {
   421→            ctx.beginPath();
   422→            ctx.arc(activeNodeObj.x, activeNodeObj.y, pulseRadius, 0, Math.PI * 2);
   423→            ctx.strokeStyle = `rgba(168, 85, 247, ${pulseAlpha})`;
   424→            ctx.lineWidth = 2;
   425→            ctx.stroke();
   426→            const pulse2 = ((time * 80 + 100) % 200);
   427→            const pulse2Alpha = Math.max(0, 1 - pulse2 / 200) * 0.08;
   428→            if (pulse2Alpha > 0.01) {
   429→              ctx.beginPath();
   430→              ctx.arc(activeNodeObj.x, activeNodeObj.y, pulse2, 0, Math.PI * 2);
   431→              ctx.strokeStyle = `rgba(6, 214, 160, ${pulse2Alpha})`;
   432→              ctx.lineWidth = 1;
   433→              ctx.stroke();
   434→            }
   435→          }
   436→        }
   437→      }
   438→
   439→      // ---- Draw connections with edge weights and curves ----

   445→
   446→          const edgeKey = [n.id, connId].sort().join('::');

   449→
   450→          const isActiveConn = (n.id === activeNode || target.id === activeNode);
   451→          const isHoveredConn = hoveredRef && (n.id === hoveredRef || target.id === hoveredRef);
   452→          const weight = n.edgeWeights[connId] || target.edgeWeights[n.id] || 0.5;
   453→          const isCurved = edgeIsCurved.get(edgeKey) || false;
   454→
   455→          const dx = target.x - n.x;
   456→          const dy = target.y - n.y;
   457→          const dist = Math.max(1, Math.sqrt(dx * dx + dy * dy));
   458→
   459→          if (isActiveConn) {
   460→            // ---- Quantum entanglement visualization ----
   461→            // Dashed animated line
   462→            ctx.beginPath();
   463→            if (isCurved) {
   464→              const midX = (n.x + target.x) / 2;
   465→              const midY = (n.y + target.y) / 2;
   466→              const perpX = -(target.y - n.y) / dist * 30;
   467→              const perpY = (target.x - n.x) / dist * 30;
   468→              ctx.moveTo(n.x, n.y);
   469→              ctx.quadraticCurveTo(midX + perpX, midY + perpY, target.x, target.y);
   470→            } else {
   471→              ctx.moveTo(n.x, n.y);
   472→              ctx.lineTo(target.x, target.y);
   473→            }
   474→            ctx.strokeStyle = `${n.color}33`;
   475→            ctx.lineWidth = 0.5;
   476→            ctx.setLineDash([3, 5]);
   477→            ctx.lineDashOffset = -time * 80;
   478→            ctx.stroke();
   479→            ctx.setLineDash([]);
   480→
   481→            // Glow line
   482→            ctx.beginPath();
   483→            if (isCurved) {
   484→              const midX = (n.x + target.x) / 2;
   485→              const midY = (n.y + target.y) / 2;
   486→              const perpX = -(target.y - n.y) / dist * 30;
   487→              const perpY = (target.x - n.x) / dist * 30;
   488→              ctx.moveTo(n.x, n.y);
   489→              ctx.quadraticCurveTo(midX + perpX, midY + perpY, target.x, target.y);
   490→            } else {
   491→              ctx.moveTo(n.x, n.y);
   492→              ctx.lineTo(target.x, target.y);
   493→            }
   494→            ctx.strokeStyle = `${n.color}44`;
   495→            ctx.lineWidth = 0.8 + weight * 0.8;
   496→            ctx.stroke();
   497→
   498→            // Bidirectional flowing particles (entanglement particles)
   499→            const particleCount = 3;
   500→            for (let pi = 0; pi < particleCount; pi++) {
   501→              // Forward particle
   502→              const fwdPos = ((time * 1.5 + pi / particleCount) % 1);
   503→              let fwdX: number, fwdY: number;
   504→              if (isCurved) {
   505→                const midX = (n.x + target.x) / 2;
   506→                const midY = (n.y + target.y) / 2;
   507→                const perpX = -(target.y - n.y) / dist * 30;
   508→                const perpY = (target.x - n.x) / dist * 30;
   509→                const t = fwdPos;
   510→                fwdX = (1 - t) * (1 - t) * n.x + 2 * (1 - t) * t * (midX + perpX) + t * t * target.x;
   511→                fwdY = (1 - t) * (1 - t) * n.y + 2 * (1 - t) * t * (midY + perpY) + t * t * target.y;
   512→              } else {
   513→                fwdX = n.x + dx * fwdPos;
   514→                fwdY = n.y + dy * fwdPos;
   515→              }
   516→              ctx.beginPath();
   517→              ctx.arc(fwdX, fwdY, 1.8, 0, Math.PI * 2);
   518→              ctx.fillStyle = n.color;
   519→              ctx.fill();
   520→              ctx.beginPath();
   521→              ctx.arc(fwdX, fwdY, 4, 0, Math.PI * 2);
   522→              ctx.fillStyle = n.color + '22';
   523→              ctx.fill();
   524→
   525→              // Reverse particle
   526→              const revPos = ((time * 1.2 + pi / particleCount + 0.5) % 1);
   527→              let revX: number, revY: number;
   528→              if (isCurved) {
   529→                const midX = (n.x + target.x) / 2;
   530→                const midY = (n.y + target.y) / 2;
   531→                const perpX = -(target.y - n.y) / dist * 30;
   532→                const perpY = (target.x - n.x) / dist * 30;
   533→                const t = revPos;
   534→                revX = (1 - t) * (1 - t) * n.x + 2 * (1 - t) * t * (midX + perpX) + t * t * target.x;
   535→                revY = (1 - t) * (1 - t) * n.y + 2 * (1 - t) * t * (midY + perpY) + t * t * target.y;
   536→              } else {
   537→                revX = target.x - dx * revPos;
   538→                revY = target.y - dy * revPos;
   539→              }
   540→              ctx.beginPath();
   541→              ctx.arc(revX, revY, 1.3, 0, Math.PI * 2);
   542→              ctx.fillStyle = target.color + 'cc';
   543→              ctx.fill();
   544→              ctx.beginPath();
   545→              ctx.arc(revX, revY, 3, 0, Math.PI * 2);
   546→              ctx.fillStyle = target.color + '18';
   547→              ctx.fill();
   548→            }
   549→          } else if (isHoveredConn) {
   550→            ctx.beginPath();
   551→            if (isCurved) {
   552→              const midX = (n.x + target.x) / 2;
   553→              const midY = (n.y + target.y) / 2;
   554→              const perpX = -(target.y - n.y) / dist * 25;
   555→              const perpY = (target.x - n.x) / dist * 25;
   556→              ctx.moveTo(n.x, n.y);
   557→              ctx.quadraticCurveTo(midX + perpX, midY + perpY, target.x, target.y);
   558→            } else {
   559→              ctx.moveTo(n.x, n.y);
   560→              ctx.lineTo(target.x, target.y);
   561→            }
   562→            ctx.strokeStyle = `${n.color}55`;
   563→            ctx.lineWidth = 0.8 + weight * 0.8;
   564→            ctx.stroke();
   565→          } else {
   566→            // Normal edge with weight-based thickness
   567→            ctx.beginPath();
   568→            if (isCurved) {
   569→              const midX = (n.x + target.x) / 2;
   570→              const midY = (n.y + target.y) / 2;
   571→              const perpX = -(target.y - n.y) / dist * 20;
   572→              const perpY = (target.x - n.x) / dist * 20;
   573→              ctx.moveTo(n.x, n.y);
   574→              ctx.quadraticCurveTo(midX + perpX, midY + perpY, target.x, target.y);
   575→            } else {
   576→              ctx.moveTo(n.x, n.y);
   577→              ctx.lineTo(target.x, target.y);
   578→            }
   579→            const alpha = isActive ? 0.12 : 0.06;
   580→            ctx.strokeStyle = `rgba(136, 136, 170, ${alpha + weight * 0.06})`;
   581→            ctx.lineWidth = 0.3 + weight * 0.6;
   582→            ctx.stroke();
   583→          }
   584→        });
   585→      });
   586→
   587→      // ---- Draw nodes with sparklines ----
   588→      hoveredRef = null;
   589→      nodes.forEach(n => {
   590→        const dx = n.x - mouseRef.current.x;
   591→        const dy = n.y - mouseRef.current.y;
   592→        const isHovered = Math.sqrt(dx * dx + dy * dy) < n.size + 5;
   593→        if (isHovered) hoveredRef = n.id;
   594→
   595→        const pulse = Math.sin(n.pulsePhase) * 0.2 + 0.8;
   596→        const drawSize = n.active ? n.size * 1.3 : isHovered ? n.size * 1.15 : n.size * pulse;
   597→
   598→        // Active glow
   599→        if (n.active || isHovered) {
   600→          ctx.beginPath();
   601→          ctx.arc(n.x, n.y, drawSize + 15, 0, Math.PI * 2);
   602→          const glowGrad = ctx.createRadialGradient(n.x, n.y, drawSize, n.x, n.y, drawSize + 15);
   603→          glowGrad.addColorStop(0, n.color + '30');
   604→          glowGrad.addColorStop(1, n.color + '00');
   605→          ctx.fillStyle = glowGrad;
   606→          ctx.fill();
   607→        }
   608→
   609→        // Node circle
   610→        ctx.beginPath();
   611→        ctx.arc(n.x, n.y, drawSize, 0, Math.PI * 2);
   612→        const nodeGrad = ctx.createRadialGradient(n.x - drawSize * 0.3, n.y - drawSize * 0.3, 0, n.x, n.y, drawSize);
   613→        nodeGrad.addColorStop(0, n.color + (n.active ? 'ee' : isHovered ? 'cc' : '88'));
   614→        nodeGrad.addColorStop(1, n.color + (n.active ? '44' : '22'));
   615→        ctx.fillStyle = nodeGrad;
   616→        ctx.fill();
   617→        ctx.strokeStyle = n.color + (n.active ? 'ff' : isHovered ? 'aa' : '55');
   618→        ctx.lineWidth = n.active ? 2 : 1;
   619→        ctx.stroke();
   620→
   621→        // Label
   622→        ctx.fillStyle = n.active || isHovered ? '#ffffff' : '#c0b8d0';
   623→        ctx.font = `${n.active ? 'bold ' : ''}${Math.max(8, n.size * 0.55)}px monospace`;
   624→        ctx.textAlign = 'center';
   625→        ctx.textBaseline = 'top';
   626→        ctx.fillText(n.label, n.x, n.y + drawSize + 5);
   627→
   628→        // ---- Sparkline (activity history) below label ----
   629→        const sparklineY = n.y + drawSize + 5 + Math.max(8, n.size * 0.55) + 4;
   630→        const sparklineW = n.size * 2.2;
   631→        const sparklineH = 6;
   632→        const sparklineX = n.x - sparklineW / 2;
   633→        const history = n.activityHistory;
   634→        if (history.length >= 2) {
   635→          ctx.beginPath();
   636→          for (let hi = 0; hi < history.length; hi++) {
   637→            const hx = sparklineX + (hi / (history.length - 1)) * sparklineW;
   638→            const hy = sparklineY + sparklineH - history[hi] * sparklineH;
   639→            if (hi === 0) ctx.moveTo(hx, hy);
   640→            else ctx.lineTo(hx, hy);
   641→          }
   642→          const sparkAlpha = n.active ? 0.7 : isHovered ? 0.5 : 0.25;
   643→          ctx.strokeStyle = n.color + (n.active ? 'bb' : isHovered ? '88' : '44');
   644→          ctx.lineWidth = 1;
   645→          ctx.stroke();
   646→
   647→          // Sparkline fill
   648→          ctx.lineTo(sparklineX + sparklineW, sparklineY + sparklineH);
   649→          ctx.lineTo(sparklineX, sparklineY + sparklineH);
   650→          ctx.closePath();
   651→          ctx.fillStyle = n.color + Math.floor(sparkAlpha * 0.3 * 255).toString(16).padStart(2, '0');
   652→          ctx.fill();
   653→        }
   654→      });
   655→
   656→      setHoveredNode(hoveredRef);
   657→      animRef.current = requestAnimationFrame(draw);
   658→    };
   659→
   660→    draw();
   661→
   662→    return () => {
   663→      cancelAnimationFrame(animRef.current);
   664→      window.removeEventListener('resize', resize);
   665→      canvas.removeEventListener('mousemove', handleMouseMove);
   666→      canvas.removeEventListener('mousedown', handleMouseDown);
   667→      canvas.removeEventListener('mouseup', handleMouseUp);
   668→      canvas.removeEventListener('mouseleave', handleMouseUp);
   669→    };
   670→  }, [isActive, activeNode, onNodeClick]);
   671→
   672→  return (
   673→    <canvas
   674→      ref={canvasRef}
   675→      className="w-full h-full block"
   676→      style={{ cursor: hoveredNode ? 'grab' : 'default' }}
   677→    />
   678→  );
   679→}
