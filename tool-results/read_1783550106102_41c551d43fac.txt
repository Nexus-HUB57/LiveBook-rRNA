    60тЖТ    60тЖТfunction getEmotionalIntensity(node: StoryNode): number {
    61тЖТ    61тЖТ  const depthBase = [0.15, 0.32, 0.48, 0.62, 0.78, 0.92];
    62тЖТ    62тЖТ  const eraMod = node.era === 'atemporal' ? 0.1 : node.era === '2077' ? 0.07 : 0.0;
    63тЖТ    63тЖТ  const hash = parseInt(node.id.slice(1), 10);
    64тЖТ    64тЖТ  const variation = Math.sin(hash * 2.7) * 0.06;
    65тЖТ    65тЖТ  return Math.min(1, Math.max(0, depthBase[node.depth] + eraMod + variation));
    66тЖТ    66тЖТ}

    68тЖТ    68тЖТfunction polarToCart(cx: number, cy: number, r: number, angleDeg: number) {
    69тЖТ    69тЖТ  const rad = (angleDeg * Math.PI) / 180;
    70тЖТ    70тЖТ  return { x: cx + r * Math.cos(rad), y: cy + r * Math.sin(rad) };
    71тЖТ    71тЖТ}

    73тЖТ    73тЖТfunction describeArc(cx: number, cy: number, r: number, startDeg: number, endDeg: number): string {
    74тЖТ    74тЖТ  const start = polarToCart(cx, cy, r, startDeg);
    75тЖТ    75тЖТ  const end = polarToCart(cx, cy, r, endDeg);
    76тЖТ    76тЖТ  const sweep = endDeg - startDeg;
    77тЖТ    77тЖТ  const large = sweep > 180 ? 1 : 0;
    78тЖТ    78тЖТ  return `M ${start.x} ${start.y} A ${r} ${r} 0 ${large} 1 ${end.x} ${end.y}`;
    79тЖТ    79тЖТ}
