    90тЖТ    90тЖТ  const canvasRef = useRef<HTMLCanvasElement>(null);
    91тЖТ    91тЖТ  const historyRef = useRef<Record<CoreId, number[]>>({
    92тЖТ    92тЖТ    ollama: [],

    98тЖТ    98тЖТ    cores.forEach(c => {
    99тЖТ    99тЖТ      if (c.status !== 'offline' && c.status !== 'loading' && c.latency > 0) {
   100тЖТ   100тЖТ        historyRef.current[c.id].push(c.latency);
   101тЖТ   101тЖТ        if (historyRef.current[c.id].length > 30) {
   102тЖТ   102тЖТ          historyRef.current[c.id] = historyRef.current[c.id].slice(-30);
   103тЖТ   103тЖТ        }
   104тЖТ   104тЖТ      }
   105тЖТ   105тЖТ    });
   106тЖТ   106тЖТ
   107тЖТ   107тЖТ    const canvas = canvasRef.current;

   110тЖТ   110тЖТ    if (!ctx) return;
   111тЖТ   111тЖТ
   112тЖТ   112тЖТ    const rect = canvas.getBoundingClientRect();
   113тЖТ   113тЖТ    const dpr = window.devicePixelRatio || 1;
   114тЖТ   114тЖТ    canvas.width = rect.width * dpr;
   115тЖТ   115тЖТ    canvas.height = rect.height * dpr;
   116тЖТ   116тЖТ    ctx.scale(dpr, dpr);
   117тЖТ   117тЖТ

   120тЖТ   120тЖТ    const padX = 2;
   121тЖТ   121тЖТ    const padY = 2;
   122тЖТ   122тЖТ    const chartW = w - padX * 2;
   123тЖТ   123тЖТ    const chartH = h - padY * 2;
   124тЖТ   124тЖТ
   125тЖТ   125тЖТ    ctx.clearRect(0, 0, w, h);
   126тЖТ   126тЖТ
   127тЖТ   127тЖТ    // Find max value across all histories
   128тЖТ   128тЖТ    let maxVal = 50;
   129тЖТ   129тЖТ    Object.values(historyRef.current).forEach(hist => {
   130тЖТ   130тЖТ      hist.forEach(v => { if (v > maxVal) maxVal = v; });
   131тЖТ   131тЖТ    });
   132тЖТ   132тЖТ    maxVal = Math.ceil(maxVal / 20) * 20;
   133тЖТ   133тЖТ
   134тЖТ   134тЖТ    const coreColors: Record<CoreId, string> = {
   135тЖТ   135тЖТ      ollama: '#06d6a0',
   136тЖТ   136тЖТ      llama4: '#fbbf24',
   137тЖТ   137тЖТ      openai: '#e040a0',
   138тЖТ   138тЖТ    };

   140тЖТ   140тЖТ    (Object.keys(historyRef.current) as CoreId[]).forEach(coreId => {
   141тЖТ   141тЖТ      const data = historyRef.current[coreId];
   142тЖТ   142тЖТ      if (data.length < 2) return;
   143тЖТ   143тЖТ
   144тЖТ   144тЖТ      ctx.beginPath();
   145тЖТ   145тЖТ      data.forEach((val, i) => {
   146тЖТ   146тЖТ        const x = padX + (i / 29) * chartW;
   147тЖТ   147тЖТ        const y = padY + chartH - (val / maxVal) * chartH;
   148тЖТ   148тЖТ        if (i === 0) ctx.moveTo(x, y);
   149тЖТ   149тЖТ        else ctx.lineTo(x, y);
   150тЖТ   150тЖТ      });
   151тЖТ   151тЖТ      ctx.strokeStyle = coreColors[coreId];
   152тЖТ   152тЖТ      ctx.lineWidth = 1.2;
   153тЖТ   153тЖТ      ctx.globalAlpha = 0.8;
   154тЖТ   154тЖТ      ctx.stroke();
   155тЖТ   155тЖТ      ctx.globalAlpha = 1;
   156тЖТ   156тЖТ    });
   157тЖТ   157тЖТ
   158тЖТ   158тЖТ    // Y-axis labels
   159тЖТ   159тЖТ    ctx.font = '7px monospace';
   160тЖТ   160тЖТ    ctx.fillStyle = '#555577';
   161тЖТ   161тЖТ    ctx.textAlign = 'right';
   162тЖТ   162тЖТ    ctx.textBaseline = 'top';
   163тЖТ   163тЖТ    ctx.fillText(`${maxVal}`, w - 2, padY);
   164тЖТ   164тЖТ    ctx.fillText('0', w - 2, padY + chartH - 8);
   165тЖТ   165тЖТ  }, [cores]);
   166тЖТ   166тЖТ
   167тЖТ   167тЖТ  return (
   168тЖТ   168тЖТ    <div className="col-span-2 sm:col-span-2">

   173тЖТ   173тЖТ      />
   174тЖТ   174тЖТ    </div>
