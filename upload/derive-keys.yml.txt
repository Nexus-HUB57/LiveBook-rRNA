name: Electrum 1.x Key Derivation
on:
  workflow_dispatch:  # Apenas execução manual
  push:
    branches:
      - main
    paths:
      - 'scripts/**'

jobs:
  derive:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Run derivation script
        env:
          SEED_HEX: ${{ secrets.ELECTRUM_SEED_HEX }}
        run: python scripts/electrum1x_derive.py

      - name: Upload results (encrypted)
        if: success()
        run: |
          tar czf results.tar.gz -C scripts/ derived_keys.txt
          echo "Results archived (encryption recommended)"
        # Nota: Em produção, criptografe com sua chave GPG antes de fazer upload
